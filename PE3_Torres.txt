Vowels = ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'O', 'U']

    Letter = input("Enter any letter: ")
    if Letter in Vowels:

        break
    else:
        print("That is not a vowel. Try Again!")
        continue
